## Usage

### Import

```jsx
import { DtLink } from '@dialpad/handtet';
```

### Default

```jsx
<dt-link
  href="#"
>
Link
</dt-link>
```

### With Kind

```jsx
<dt-link
  href="#"
  kind="danger | warning | success | muted"
>
Link
</dt-link>
```

### Disabled

```jsx
<dt-link
  href="#"
  disabled
>
Link
</dt-link>
```

### Inverted

```jsx
<dt-link
  href="#"
  inverted
>
Link
</dt-link>
```

### With Disabled and Inverted

```jsx
<dt-link
  href="#"
  disabled
  inverted
>
Link
</dt-link>
```

### With Inverted and Danger

```jsx
<dt-link
  href="#"
  inverted
  kind="danger"
>
Link
</dt-link>
```
