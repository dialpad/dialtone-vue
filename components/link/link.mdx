## Usage

### Import

```jsx
import { DtLink } from '@dialpad/handtet';
```

### Default

```jsx
<dt-link
  href="#"
>
Link
</dt-link>
```

### With Kind

```jsx
<dt-link
  href="#"
  kind="danger | warning | success | muted | inverted"
>
Link
</dt-link>
```
