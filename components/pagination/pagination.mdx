## Usage

### Import

```jsx
import { DtPagination } from '@dialpad/dialtone-vue';
```

### Basic

```html
<dt-pagination
  :total-pages="20"
  :prev-aria-label="$i18n('previous')"
  :next-aria-label="$i18n('next')"
/>
```

### With active-page

```html
<dt-pagination
  :total-pages="10"
  :active-page="5"
  :prev-aria-label="$i18n('previous')"
  :next-aria-label="$i18n('next')"
/>
```

### With max-visible

```html
<dt-pagination
  :total-pages="15"
  :max-visible="7"
  :prev-aria-label="$i18n('previous')"
  :next-aria-label="$i18n('next')"
/>
```
