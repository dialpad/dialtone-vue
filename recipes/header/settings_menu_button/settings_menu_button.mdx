## Usage

### Import

```jsx
import { DtRecipeSettingsMenuButton } from '@dialpad/dialtone-vue';
```

### Default

```html
<dt-recipe-settings-menu-button
  :aria-label="ariaLabel"
  @click="onClick"
/>
```

### Update Available

```html
<dt-recipe-settings-menu-button
  :update-available="true"
  :aria-label="ariaLabel"
  @click="onClick"
>
  Update
</dt-recipe-settings-menu-button>
```
